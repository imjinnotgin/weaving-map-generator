<!DOCTYPE html>

<html lang="en">

  <head>

    <meta charset="UTF-8" />

    <title>Weaving Draft Map Generator</title>

    <style>

      body { margin: 0; overflow: hidden; }

      #map { position: absolute; top: 0; left: 0; width: 100%; height: 100%; z-index: 0; }

      canvas { position: absolute; top: 0; left: 0; z-index: 1; pointer-events: none; }

    </style>

    <!-- Leaflet.js for the map -->

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.3/dist/leaflet.css" />

    <script src="https://unpkg.com/leaflet@1.9.3/dist/leaflet.js"></script>

    <!-- p5.js -->

    <script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.9.0/p5.min.js"></script>

  </head>

  <body>

    <div id="map"></div>

    <script>

      let map;

      let canvas;


      function setup() {

        canvas = createCanvas(windowWidth, windowHeight);

        canvas.position(0, 0);

        canvas.style('pointer-events', 'none');

        noLoop();

        setupMap();

      }


      function setupMap() {

        map = L.map('map').setView([37.5665, 126.9780], 13); // 서울 기준


        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

          attribution: '&copy; OpenStreetMap contributors'

        }).addTo(map);


        map.on('click', function(e) {

          const lat = e.latlng.lat;

          const lon = e.latlng.lng;

          drawWeavingPattern(lat, lon);

        });

      }


      function drawWeavingPattern(lat, lon) {

        clear();

        const size = 700;

        const cols = 20;

        const rows = 20;

        const cellSize = size / cols;

        const seed = int((lat + lon) * 10000);

        randomSeed(seed);

        push();

        translate((width - size) / 2, (height - size) / 2);

        noFill();

        stroke(0);

        strokeWeight(1);

        rect(0, 0, size, size);

        for (let y = 0; y < rows; y++) {

          for (let x = 0; x < cols; x++) {

            if (random() > 0.5) {

              line(x * cellSize, y * cellSize, (x + 1) * cellSize, (y + 1) * cellSize);

            } else {

              line((x + 1) * cellSize, y * cellSize, x * cellSize, (y + 1) * cellSize);

            }

          }

        }

        pop();

      }


      function windowResized() {

        resizeCanvas(windowWidth, windowHeight);

      }

    </script>

  </body>
